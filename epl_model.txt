model
{
 for(i in 1:nTeams){
   y[i] ~ dnorm(mu[i], 1/sigma_2)
   mu[i] <- alpha[i] * scored[i] + beta[i] * conceded[i]
   alpha[i] ~ dnorm(alpha_0, 1/sigma_a2)
   beta[i] ~ dnorm(beta_0, 1/sigma_b2)

 }

 alpha_0 ~ dnorm(0, 1/100)
 beta_0 ~ dnorm(0, 1/100)


 sigma_2 ~ dgamma(1,1)
 sigma_a2 ~ dgamma(1,1)
 sigma_b2 ~ dgamma(1,1)

}
